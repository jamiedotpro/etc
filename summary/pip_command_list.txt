# venv 환경이 필요한 프로젝트 폴더로 이동 후 아래 명령어 실행
python -m venv .venv

# 환경 활성화
.venv\Scripts\activate

# 비활성화
deactivate

# 위의 activate 가 실행되지 않을 때 아래의 명령어 입력해서 보안 설정 변경
Set-ExecutionPolicy Unrestricted -Scope CurrentUser
# 그래도 안되면 아래 명령어 입력
Set-ExecutionPolicy Unrestricted -Force
